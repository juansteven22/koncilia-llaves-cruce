cd koncilia-llaves-cruce

# Crear proyecto Web API .NET 8 con Swagger (útil en local)

dotnet new webapi -n Etiquetado --framework net8.0

dotnet add package Microsoft.Data.SqlClient
dotnet add package Dapper

# Agregar el proyecto a tu solución
cd ..
dotnet sln ..\KonciliaLlavesCruce.sln add Etiquetado\Etiquetado.csproj
